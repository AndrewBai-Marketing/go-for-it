\section{Introduction}
%------------------------------------------

\begin{frame}{The Question}
\begin{itemize}
\item \textbf{Setting:} NFL coaches make high-stakes decisions with well-defined objectives
\item \textbf{Classic finding:} Romer (2006) showed coaches are systematically too conservative on fourth down
\item \textbf{The puzzle:} 20 years of publicly available research, billions of dollars at stake
\item \textbf{Our question:} Have coaches improved? Are they learning?
\end{itemize}

\vspace{0.5cm}
\pause
\begin{block}{Preview}
Fourth down: No improvement. Two-point conversions: Yes, significant improvement.
\end{block}
\end{frame}

\section{Fourth Down Decisions}
%------------------------------------------

\subsection{The Model}

\begin{frame}{The Decision Problem}
\textbf{State:} Score differential $\Delta$, time remaining $\tau$, field position $x$, yards to go $d$

\vspace{0.3cm}
\textbf{Actions:} $\mathcal{A} = \{\texttt{go}, \texttt{punt}, \texttt{fg}\}$

\vspace{0.3cm}
\textbf{Objective:} Maximize win probability
\[
a^* = \arg\max_{a \in \mathcal{A}} \E[W \mid a, s]
\]

\pause
\vspace{0.3cm}
\textbf{Key innovation:} Bayesian framework with hierarchical effects
\begin{itemize}
\item Offensive team conversion ability $\gamma_j^{\text{off}}$
\item Defensive team stopping ability $\delta_k^{\text{def}}$
\item Kicker-specific field goal accuracy
\end{itemize}
\end{frame}

\begin{frame}{Can the Model Capture the Coach's Information?}
\textbf{Selection-on-observables argument:} We condition on what the coach knows

\vspace{0.5cm}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Model conditions on:}
\begin{itemize}
\item Game state (score, time, position)
\item Offensive team quality
\item Defensive team quality
\item Kicker ability
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Direct test:}\\[0.2cm]
When coaches override the model, do they convert more often?
\begin{itemize}
\item 4th \& short: +3.0 pp (some info)
\item 4th \& medium: +0.4 pp (none)
\item 4th \& long: $-9.3$ pp (overconfident)
\end{itemize}
\end{column}
\end{columns}

\pause
\vspace{0.5cm}
\begin{alertblock}{Takeaway}
Coaches don't outperform the model $\Rightarrow$ suboptimality is behavioral, not informational
\end{alertblock}
\end{frame}

\subsection{Results}

\begin{frame}{Real-Time Knowability}
\textbf{Objection:} ``You can't judge coaches using hindsight''

\vspace{0.3cm}
\textbf{Test:} Expanding window analysis (7-year minimum training)
\begin{itemize}
\item Train model on data available \textit{before} each season
\item Compare ex ante vs.\ ex post recommendations
\end{itemize}

\pause
\vspace{0.5cm}
\begin{center}
\begin{tabular}{lc}
\toprule
\textbf{Finding} & \textbf{Value} \\
\midrule
Ex ante/ex post agreement & 96.5\% \\
Plays analyzed & 71,786 \\
Years tested & 2006--2024 \\
\bottomrule
\end{tabular}
\end{center}

\pause
\vspace{0.3cm}
\begin{block}{Result}
96.5\% of optimal decisions were \textbf{knowable in real-time}
\end{block}
\end{frame}

\begin{frame}{How Often Are Coaches Optimal?}
\begin{center}
\begin{tabular}{lrr}
\toprule
\textbf{Category} & \textbf{Count} & \textbf{\%} \\
\midrule
\textbf{Correct} (matched model) & 57,796 & \textbf{80.5\%} \\
Difference of opinion ($<$2\% WP) & 11,558 & 16.1\% \\
Questionable (2--5\% WP) & 1,758 & 2.4\% \\
Clear mistake ($>$5\% WP) & 674 & \textbf{0.9\%} \\
\midrule
\textbf{Total} & \textbf{71,786} & 100\% \\
\bottomrule
\end{tabular}
\end{center}

\pause
\vspace{0.5cm}
\begin{block}{Key Finding}
83\% of ``mistakes'' are actually close calls. Only 0.9\% are clear errors.
\end{block}
\end{frame}

\section{Two-Point Conversion Decisions}
%------------------------------------------

\subsection{The Model}

\begin{frame}{The Two-Point Decision}
After a touchdown: PAT (1 point, 94\% success) or 2pt (2 points, 48\% success)

\vspace{0.3cm}
\textbf{Expected points nearly identical:}
\[
\E[\text{PAT}] = 0.94 \quad \text{vs.} \quad \E[\text{2pt}] = 0.96
\]

\pause
\vspace{0.3cm}
\textbf{The difference is entirely \textit{variance}:}
\begin{itemize}
\item PAT: High probability, low variance
\item 2pt: Lower probability, higher variance
\end{itemize}

\pause
\vspace{0.3cm}
\begin{block}{Simple Rule}
\textbf{Trailing?} Go for 2 (variance helps when behind)\\
\textbf{Leading?} Kick PAT (variance hurts when ahead)
\end{block}
\end{frame}

\subsection{Results}

\begin{frame}{Two-Point Coach Optimality}
\begin{center}
\begin{tabular}{lrr}
\toprule
\textbf{Category} & \textbf{Count} & \textbf{\%} \\
\midrule
\textbf{Correct} & 8,211 & 58.6\% \\
Difference of opinion ($<$2\% WP) & 5,811 & 41.4\% \\
Questionable (2--5\% WP) & 0 & 0.0\% \\
Clear mistake ($>$5\% WP) & 1 & 0.007\% \\
\midrule
\textbf{Total} & \textbf{14,023} & 100\% \\
\bottomrule
\end{tabular}
\end{center}

\pause
\vspace{0.5cm}
\begin{block}{Key Finding}
Virtually all deviations are close calls. Only 1 play out of 14,023 was a clear mistake.
\end{block}

\vspace{0.3cm}
{\small This reflects the nature of the decision: outcomes differ by only 1 point.}
\end{frame}

\section{Comparison and Conclusion}
%------------------------------------------

\begin{frame}{Fourth Down vs.\ Two-Point Comparison}
\begin{center}
\begin{tabular}{lcc}
\toprule
\textbf{Metric} & \textbf{Fourth Down} & \textbf{Two-Point} \\
\midrule
Coach optimality rate & 80.5\% & 58.6\% \\
Clear mistakes (\% of all) & 0.9\% & 0.007\% \\
Maximum WP margin & $>$30\% & $\sim$5\% \\
Team effects matter? & Yes ($\tau \approx 0.15$) & No ($\tau \approx 0.03$) \\
\textbf{Learning trend} & \textbf{None} & \textbf{+1 pp/year} \\
\bottomrule
\end{tabular}
\end{center}

\pause
\vspace{0.5cm}
\textbf{Paradox:} Fourth down decisions matter more but coaches perform \textit{better} at them.

\vspace{0.3cm}
Two-point decisions matter less but coaches are \textit{worse}---and improving.

\pause
\vspace{0.3cm}
\begin{alertblock}{Takeaway}
Decision \textit{complexity}, not stakes, drives coaching errors
\end{alertblock}
\end{frame}

\begin{frame}{Conclusion: Is Management Learning?}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Fourth Down:}
\begin{itemize}
\item 80.5\% optimal decisions
\item 83\% of mistakes are close calls
\item 96.5\% knowable in real-time
\item \alert{No improvement over 20 years}
\item Shifted from too conservative to too aggressive
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Two-Point:}
\begin{itemize}
\item 58.6\% optimal decisions
\item Virtually all deviations are close calls
\item \textbf{+1 pp/year improvement}
\item Still 35 pp too conservative
\item But stakes are low
\end{itemize}
\end{column}
\end{columns}

\pause
\vspace{0.5cm}
\begin{block}{Bottom Line}
Coaches are learning at the simple decision (2pt) but not the complex one (4th down).\\
The ``analytics revolution'' changed behavior but not accuracy.
\end{block}
\end{frame}

\begin{frame}{Key Takeaways}
\begin{enumerate}
\item \textbf{Coaches are mostly right:} 80\% optimal on 4th down, 59\% on 2pt
\vspace{0.2cm}
\item \textbf{Most ``mistakes'' are close calls:} 83\% of deviations have $<$2\% WP margin
\vspace{0.2cm}
\item \textbf{The information exists:} 96.5\% of optimal decisions were knowable in real-time
\vspace{0.2cm}
\item \textbf{No improvement on 4th down:} Despite 20 years of research and increased aggression
\vspace{0.2cm}
\item \textbf{Improvement on 2pt:} +1 pp/year (simpler decision $\Rightarrow$ easier to learn)
\vspace{0.2cm}
\item \textbf{Free wins still on the table:} Teams lose $\sim$0.34 expected wins/season from 4th down errors alone
\end{enumerate}
\end{frame}
